from datetime import datetime, timedelta


def normalize_busy_intervals(busy_intervals):
    """Sort and merge overlapping or adjacent busy intervals."""
    
    if not busy_intervals:
        return []

    busy_intervals = sorted(busy_intervals, key=lambda x: x[0])
    merged = [busy_intervals[0]]

    for start, end in busy_intervals[1:]:
        last_start, last_end = merged[-1]

        if start <= last_end:  # overlap OR adjacency
            merged[-1] = (last_start, max(last_end, end))
        else:
            merged.append((start, end))

    return merged


def validate_inputs(work_start, work_end, busy_intervals, meeting_duration,
                    buffer_time, candidate_lower, candidate_upper, N):

    if work_start >= work_end:
        raise ValueError("work_start must be before work_end")

    if meeting_duration < 1:
        raise ValueError("meeting_duration must be >= 1")

    if buffer_time < 0:
        raise ValueError("buffer_time must be >= 0")

    if N < 0:
        raise ValueError("N must be >= 0")

    if (candidate_lower is None) != (candidate_upper is None):
        raise ValueError("Candidate window must have both bounds or neither")

    for start, end in busy_intervals:
        if start >= end:
            raise ValueError("Busy interval start must be before end")

        if start < work_start or end > work_end:
            raise ValueError("Busy intervals must lie within working hours")

    if candidate_lower and candidate_upper:

        if candidate_lower < work_start or candidate_upper > work_end:
            raise ValueError("Candidate window must lie within working hours")

        if candidate_lower > candidate_upper:
            raise ValueError("Invalid candidate window bounds")


def recommend_appointments(
    work_start,
    work_end,
    interval_busy,
    meeting_duration,
    buffer_time,
    candidate_lower=None,
    candidate_upper=None,
    N=1
):

    validate_inputs(
        work_start,
        work_end,
        interval_busy,
        meeting_duration,
        buffer_time,
        candidate_lower,
        candidate_upper,
        N
    )

    if N == 0:
        return []

    duration = timedelta(minutes=meeting_duration)
    buffer_delta = timedelta(minutes=buffer_time)

    busy = normalize_busy_intervals(interval_busy)

    if candidate_lower is None:
        candidate_lower = work_start
        candidate_upper = work_end

    search_start = max(work_start, candidate_lower)
    search_end = min(work_end, candidate_upper)

    results = []
    current_time = search_start

    while current_time + duration <= search_end:

        meeting_start = current_time
        meeting_end = meeting_start + duration

        valid = True

        for busy_start, busy_end in busy:

            # enforce buffer before meeting relative to busy interval
            if meeting_start < busy_end + buffer_delta and meeting_end > busy_start - buffer_delta:
                valid = False
                break

        if valid:

            if results:
                last_start, last_end = results[-1]

                if meeting_start < last_end + buffer_delta:
                    valid = False

        if valid:

            results.append((meeting_start, meeting_end))

            if len(results) == N:
                return results

            current_time = meeting_end + buffer_delta
        else:
            current_time += timedelta(minutes=1)

    return results

def suggest_slots(
    day: date,
    working_hours: TimeWindow,
    busy_intervals: List[BusyInterval],
    duration: timedelta,
    n: int,
    buffer: timedelta = timedelta(0),
    candidate_window: Optional[TimeWindow] = None
) -> List[Slot]:

    # ---------------- Input Validation ----------------

    if working_hours.start >= working_hours.end:
        raise ValueError("Working hours must satisfy start < end")

    if duration <= timedelta(0):
        raise ValueError("Meeting duration must be >= 1 minute")

    if buffer < timedelta(0):
        raise ValueError("Buffer must be >= 0")

    if n < 0:
        raise ValueError("n must be >= 0")

    if n == 0:
        return []

    if candidate_window is not None:
        if candidate_window.start >= candidate_window.end:
            raise ValueError("Candidate window must satisfy start < end")

        if (candidate_window.start < working_hours.start or
                candidate_window.end > working_hours.end):
            raise ValueError("Candidate window must lie within working hours")

    for interval in busy_intervals:
        if interval.start >= interval.end:
            raise ValueError("Busy interval start must be < end")

        if interval.start < working_hours.start or interval.end > working_hours.end:
            raise ValueError("Busy interval outside working hours")

    # ---------------- Helper Conversions ----------------

    def combine(t: time) -> datetime:
        return datetime.combine(day, t)

    work_start_dt = combine(working_hours.start)
    work_end_dt = combine(working_hours.end)

    if candidate_window:
        search_start = max(work_start_dt, combine(candidate_window.start))
        search_end = min(work_end_dt, combine(candidate_window.end))
    else:
        search_start = work_start_dt
        search_end = work_end_dt

    # ---------------- Normalize Busy Intervals ----------------

    busy_dt = [(combine(b.start), combine(b.end)) for b in busy_intervals]

    busy_dt.sort(key=lambda x: x[0])

    merged_busy = []
    for start, end in busy_dt:
        if not merged_busy:
            merged_busy.append([start, end])
        else:
            last_start, last_end = merged_busy[-1]

            if start <= last_end:  # overlap or adjacency
                merged_busy[-1][1] = max(last_end, end)
            else:
                merged_busy.append([start, end])

    merged_busy = [(s, e) for s, e in merged_busy]

    # ---------------- Slot Search ----------------

    results: List[Slot] = []
    current = search_start

    while current + duration <= search_end:

        meeting_start = current
        meeting_end = meeting_start + duration

        valid = True

        # Check conflicts with busy intervals
        for busy_start, busy_end in merged_busy:

            if (meeting_start < busy_end + buffer and
                    meeting_end > busy_start - buffer):
                valid = False
                break

        # Check buffer relative to previous meeting
        if valid and results:
            prev_start = combine(results[-1].start_time)
            prev_end = prev_start + duration

            if meeting_start < prev_end + buffer:
                valid = False

        if valid:
            results.append(Slot(start_time=meeting_start.time()))

            if len(results) == n:
                break

            current = meeting_end + buffer
        else:
            current += timedelta(minutes=1)

    return results
